New Variable Declarations

let and const

let

let defines a block level variable. It can help prevent confusion in going through source code.

Prevents errors


function test() {
    console.log(x); //undefined
    var x = 't';
    console.log(x); // t
}
test();

This focuses on the comment.

Scoping


function test2() {
    var x = 't';
    console.log(x);  //t

    if (x == 't') {
        var x = "y"
        console.log(x);  //y
    }

    console.log(x);  //y
}

test2();             

function test() {
    let x = 't';
    console.log(x); // t
      
    if (x == 't') {
        let x = "y"
        console.log(x); //y
    }
    
    console.log(x);  //t
}

test();

const

const is used to generate variables that do not change.

Using const


var pi = 3.14

function calc(d) {
    return d * pi;
}

calc(4);

/* 100 lines down */

pi = 4;  

/* 100 lines down */

calc(4);//Different output
        

const pi2 = 3.14

function calc2(d) {
    return d * pi2;
}

calc2(4);

pi = 4;  // errors

calc(4);//Different output
