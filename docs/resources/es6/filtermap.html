.forEach()

forEach loops through each item in an array and executes a function for each item.

Syntax


[1,2,3].forEach(function(element, index, array){
    //Code
});

  • element - the current iterated element.
  • index - the current zero based index
  • array - the full array of items
  • Does not return anything

Example - Simple


const names = ["Daniel","Josh","Amanda","Jennifer"];
names.forEach(function(element, index, array){
    const message = `Hello, my name is ${element} and I'm the ${index} position`;
    console.log(message);
});
//Hello, my name is Daniel and I'm the 0 position
//Hello, my name is Josh and I'm the 1 position
//Hello, my name is Amanda and I'm the 2 position
//Hello, my name is Jennifer and I'm the 3 position

Example - More Complex


const users = [
    {name:"Daniel", email:"d@yahoo.com"},
    {name:"Josh", email:""},
    {name:"Amanda", email:""},
    {name:"Jennifer", email:"a@yahoo.com"},
]
users.forEach(function(element, index, array){
    let haveEmail = "don't" 
    if (element.email !== ''){
       haveEmail = 'do'; 
    }
    const message = `Hello, my name is ${element.name} and I ${haveEmail} have an email address`;
    console.log(message);
});
        
        
//Hello, my name is Daniel and I do have an email address
//Hello, my name is Josh and I don't have an email address
//Hello, my name is Amanda and I don't have an email address
//Hello, my name is Jennifer and I do have an email address

.map()

map acts similar to forEach but it returns a whole new array.

Syntax


cosnt x = [1,2,3].map(function(e, i, a){ 
    return '';
});

  • e -> element - the current iterated element.
  • i -> index - the current zero based index
  • a -> array - the full array of items
  • Returned item gets added to new array.
  • Returns new array

Example


const users = [
    {fName:"Daniel", lName:"Worthy"},
    {fName:"Josh", lName:"Worthy"},
    {fName:"Amanda", lName:"Satterfield"},
    {fName:"Jennifer", lName:"England"},
]
const fullNames = users.map(function(element, index, array){
    return `${element.fName} ${element.lName}`;
});

console.log(fullNames);
//["Daniel Worthy", "Josh Worthy", "Amanda Satterfield", "Jennifer England"]

.filter()

Filter iterates through an array and adds the element to the new array if it returns true.

Syntax

var x = [1,2,3].filter(function(e, i, a){
    return true | false;
});

  • e -> element - the current iterated element.
  • i -> index - the current zero based index
  • a -> array - the full array of items
  • Returns new array

Example Simple


const nums = [10, 20, 33, 42, 51];

const evenNumbers = nums.filter(function(element, index, array){
    return element % 2 === 0;
});
const oddNumbers = nums.filter(function(element, index, array){
    return element % 2 !== 0;
});

console.log(`Even ${evenNumbers}`); //"Even 10,20,42"
console.log(`Odd  ${oddNumbers}`); //"Odd  33,51"

const users = [

.find()

Acts like filter, but only returns the first one found that matches the parameters. It is like EF's .FirstOrDefault()

Syntax

 var x = [1,2,3].find(function(e, i, a){
    return true | false;
});

  • e -> element - the current iterated element.
  • i -> index - the current zero based index
  • a -> array - the full array of items
  • Returns first item matching parameters

Example


const users = [
    {id:1, name:"Daniel", email:"d@yahoo.com", sex:'M'},
    {id:10, name:"Daniel", email:"d2@yahoo.com", sex:'M'},
    {id:2, name:"Josh", email:"", sex: 'M'},
    {id: 3, name:"Amanda", email:"", sex:'F'},
    {id: 4, name:"Jennifer", email:"a@yahoo.com", sex:'F'},
]

function getByName(name){
    return users.find(function(user){
       return user.name === name; 
    })
}

console.log(getByName('Daniel')) //{d@yahoo.com}
console.log(getByName('Josh'))   //{josh}
